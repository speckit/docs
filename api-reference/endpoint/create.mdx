---
title: "Create User"
api: "POST https://api.speckit.dev/v1/users"
description: "Create a new user account"
---

## Body Parameters

<ParamField body="email" type="string" required>
  User's email address (must be unique)
</ParamField>

<ParamField body="name" type="string" required>
  User's full name
</ParamField>

<ParamField body="password" type="string" required>
  User's password (minimum 8 characters)
</ParamField>

<ParamField body="profile" type="object">
  Optional profile information
  
  <Expandable title="profile">
    <ParamField body="bio" type="string">
      User's biography (max 500 characters)
    </ParamField>
    
    <ParamField body="location" type="string">
      User's location
    </ParamField>
    
    <ParamField body="website" type="string">
      User's website URL
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="settings" type="object">
  User preferences and settings
  
  <Expandable title="settings">
    <ParamField body="notifications_enabled" type="boolean" default="true">
      Whether to send email notifications
    </ParamField>
    
    <ParamField body="theme" type="string" default="light">
      UI theme preference (`light`, `dark`, or `auto`)
    </ParamField>
    
    <ParamField body="timezone" type="string" default="UTC">
      User's timezone (IANA timezone identifier)
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique identifier for the created user
</ResponseField>

<ResponseField name="email" type="string">
  User's email address
</ResponseField>

<ResponseField name="name" type="string">
  User's full name
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the user was created
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of when the user was last updated
</ResponseField>

```javascript JavaScript
const userData = {
  email: "jane.smith@example.com",
  name: "Jane Smith",
  password: "securepassword123",
  profile: {
    bio: "Product manager with a passion for user experience",
    location: "New York, NY",

  method: 'POST',
  headers: {
    'Authorization': 'Bearer your-api-key',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify(userData)
});

const newUser = await response.json();
console.log(newUser);
```

```python Python
import requests

url = "https://api.speckit.dev/v1/users"
headers = {
    "Authorization": "Bearer your-api-key",
    "Content-Type": "application/json"
}

data = {
    "email": "jane.smith@example.com",
    "name": "Jane Smith",
    "password": "securepassword123",
    "profile": {
        "bio": "Product manager with a passion for user experience",
        "location": "New York, NY",
        "website": "https://janesmith.dev"
    },
    "settings": {
        "notifications_enabled": True,
        "theme": "dark",
        "timezone": "America/New_York"
    }
}

response = requests.post(url, headers=headers, json=data)
new_user = response.json()
print(new_user)
```
